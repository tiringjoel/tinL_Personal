\begin{frame}{Outline}
 \begin{itemize}
  \item Development
  \begin{itemize}
   \item Programme auf dem \host für den \host
  \end{itemize}
  \item CrossDevelopment
  \begin{itemize}
   \item Programme auf dem \host für für den \target
  \end{itemize}
 \end{itemize}
\end{frame}

\begin{frame}{Ziel}
 \begin{itemize}
  \item CrossDevelopment (fast) gleich wie Development
  \item wegen {\bf POSIX}
 \end{itemize}
\end{frame}
